# 使い方

## .envを設定する

テンプレート
```
# Oauth プロバイダのキーなど
DISCORD_CLIENT_ID = 
DISCORD_CLIENT_SECRET = 
DISCORD_CALLBACK_URL = 

# ランダムな文字列 (64文字くらい)
Session_Key = 
Session_MaxAge = 

# ランダムな文字列 (64文字くらい)
Transaction_Store_Secret = 
Token_Secret = 

# Redis に接続するための情報
Redis_Host = 
Redis_Port = 
Redis_Password = 
Token_Redis_DB = 0

# ランダムな文字列 (64文字くらい)
JWT_Secret = 
```
